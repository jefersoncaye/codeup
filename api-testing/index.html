    API Testing com Playwright e Python¶

    Playwright não é só para UI — o APIRequestContext permite fazer requisições HTTP diretamente, no mesmo framework que você já usa para testes de interface.

    Fixture de API Request¶

    import pytest
from playwright.sync_api import APIRequestContext, Playwright


@pytest.fixture(scope="session")
def api_request(playwright: Playwright) -> APIRequestContext:
    request = playwright.request.new_context(
        base_url="https://api.exemplo.com",
        extra_http_headers={"Content-Type": "application/json"},
    )
    yield request
    request.dispose()

    Exemplo — GET com validação¶

    def test_listar_usuarios(api_request: APIRequestContext) -> None:
    response = api_request.get("/users")
    assert response.status == 200
    body = response.json()
    assert isinstance(body, list)
    assert len(body) > 0
    assert "id" in body[0]
    assert "email" in body[0]

    Exemplo — POST com autenticação¶

    def test_criar_usuario(api_request: APIRequestContext) -> None:
    payload = {"nome": "Maria", "email": "maria@teste.com", "perfil": "viewer"}
    response = api_request.post("/users", data=payload)
    assert response.status == 201
    body = response.json()
    assert body["email"] == "maria@teste.com"
    assert "id" in body
